Boucle sur release_year pour chaque type

Boucle sur les 10 dernires annes

dix_derniere_annee <- tail(sort(unique(netflix$release_year)), 10)
for (annee in dix_derniere_annee) {
  contenu_annee <- subset(netflix, release_year == annee)
  contenu_annee_type <- table(contenu_annee$type)
  barplot(contenu_annee_type, main = paste("Nombre de chaque type en", annee), xlab = "Type", ylab = "Nombre", col = c("chartreuse", "cyan"))
}
